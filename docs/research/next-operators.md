```python
# Standard ATK-buff operator pattern
_S1_TAG = "{op}_s1_atk_up"
_S1_ATK_RATIO = 1.00  # e.g. +100%
_S1_BUFF_TAG = "{op}_s1_atk"
_S1_DURATION = 30.0

def _s1_on_start(world, carrier):
    carrier.buffs.append(Buff(axis=BuffAxis.ATK, stack=BuffStack.RATIO,
                              value=_S1_ATK_RATIO, source_tag=_S1_BUFF_TAG))

def _s1_on_end(world, carrier):
    carrier.buffs = [b for b in carrier.buffs if b.source_tag != _S1_BUFF_TAG]
```
